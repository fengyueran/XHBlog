### **快速排序法**

```
const quickSort = (arr) => {
  if (arr.length <= 1) return arr;
  const pivotIndex = Math.floor(arr.length / 2);
  const pivot = arr.splice(pivotIndex, 1);
  const left = [];
  const right = [];

  for(let i = 0; i< arr.length; i++) {
    const value = arr[i];
    if (value < pivot) {
      left.push(value);
    } else {
      right.push(value);
    }
  }

  return quickSort(left).concat(pivot, quickSort(right));
}
```
